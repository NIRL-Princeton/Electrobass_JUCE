139  void syncAll()
140  {
141  /* Create a deep copy of the value tree. Pass it to the other thread. */
142  ValueTreeCopy* t = new ValueTreeCopy (sourceTree);
143  /* Create tree to tree mapping so property updates can happen quickly. */

151  void valueTreePropertyChanged (ValueTree& tree, const Identifier& property)
152  {
153  /* Send an update property message. Property removals are handled by sending
154  a null var() object.
155 
156  A property addition however requires that we replace the entire value tree
157  to avoid calling malloc on the audio thread.
158  */
159  if (! isPropertyChangedOperationLockFree (tree, property))
160  {
161  syncAll();
162  return;
163  }
164 
165  /* Create a new ValueTree that points to the same underlying data. */
166  ValueTree target = linkCache[tree];
167  /* Create a var object. Note: var manages the reference counting. */
168  var v = tree[property];
169  /* If it's an object, check that the GarbageCollector knows about it.
170  Otherwise it may end up being deleted on the critical thread.
171 
172  If you hit this assert you probably aren't using a GarbageCollectedObject
173  and instead you've got a standard ReferenceCountedObject here.
174  */
175  jassert (
176  (! v.isObject()) ||
177  (GarbageCollector::getInstance()->isInList (v.getObject()))
178  );
179  /* Pass these objects to the critical thread to update the critical thread's
180  own ValueTree. */
181  sendUpdateProperty (target, property, v);
182  }

186  bool isPropertyChangedOperationLockFree (ValueTree& tree, const Identifier& property)
187  {
188  if (tree[property].isString()) return false; /* Might be okay? */
189 
190  ValueTree t = linkCache[tree];
191 
192  if (t == ValueTree::invalid)
193  {
194  jassertfalse; /* This shouldn't happen: we should have all the nodes at least. */
195  return false;
196  }
197 
198  return (t.hasProperty (property));
199  }
